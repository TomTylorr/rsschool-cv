function smallest(arr) {
    let smallest = arr[0];
    let smallestIndex = 0;

    for (let i = 1; i < arr.length; i++) {
        if (arr[i] < smallest) {
            smallest = arr[i];
            smallestIndex = i;
        }
    }
    return smallestIndex;
}

function selectionSort(arr) {
    let newArr = [];

    for (let i = 0; arr.length; i++) {
        let min = smallest(arr);
        newArr.push(arr.splice(min, 1));
    }
    return newArr;
}
